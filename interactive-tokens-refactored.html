Usage — Combining Surface + Scale Tokens

Apply both classes to any interactive element

<!-- Card: fillNeutral surface + scale-700 -->
<div class="surface-fillNeutral scale-700">
  Card content
</div>

<!-- Button: fillColor surface + scale-500 -->
<button class="surface-fillColor scale-500">
  Buy Tickets
</button>

<!-- Icon button: borderNeutral + scale-300 -->
<button class="surface-borderNeutral scale-300">
  +
</button>

<!-- Ghost surface + card-size scale -->
<div class="surface-ghost scale-700">
  Clickable row
</div>
