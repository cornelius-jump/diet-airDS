Usage — Combining Surface + Scale Tokens

<!-- A card with fillNeutral surface + scale-700 interaction -->
<div class="surface-fillNeutral scale-700">
  Card content
</div>

<!-- A button with fillColor surface + scale-500 interaction -->
<button class="surface-fillColor scale-500">
  Buy Tickets
</button>

<!-- An icon button with borderNeutral surface + scale-300 -->
<button class="surface-borderNeutral scale-300">
  +
</button>

<!-- Ghost surface (no fill) + card-size scale -->
<div class="surface-ghost scale-700">
  Clickable row
</div>
